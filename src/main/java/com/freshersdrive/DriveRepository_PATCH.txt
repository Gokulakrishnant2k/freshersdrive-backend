    // =========================

    // Dedup key: normalized hash of (companyName, jobRole, location).
    // Unique so the DB itself enforces no duplicate postings.
    @Column(unique = true, length = 64)
    private String fingerprint;

    // Where this drive came from. Defaults to MANUAL for anything saved
    // through the existing AddDrive flow (Builder.Default handles that).
    @Enumerated(EnumType.STRING)
    @Builder.Default
    private DriveSource source = DriveSource.MANUAL;

    // Review state for AI-discovered drives. Manually-added drives are
    // APPROVED immediately since a human typed them in directly.
    @Enumerated(EnumType.STRING)
    @Builder.Default
    private ReviewStatus reviewStatus = ReviewStatus.APPROVED;

    // True if `deadline` was a guessed fallback (driveDate + 30 days)
    // because AI extraction couldn't find a real deadline on the page.
    // Surface this in the admin review UI so deadlines get checked.
    @Builder.Default
    private Boolean deadlineGuessed = false;

// 3) IMPORTANT: because Drive uses @Builder (Lombok), the @Builder.Default
//    annotations above are required for the defaults to actually apply
//    when using Drive.builder()...build() WITHOUT explicitly setting that
//    field. Without @Builder.Default, Lombok's builder silently ignores
//    the field initializer and the field is null/false regardless of what
//    you wrote above. This matters for your existing AddDrive flow, which
//    presumably uses Drive.builder() too — double check AddDrive's save
//    path still compiles and sets reviewStatus correctly (or doesn't set
//    it, relying on this default of APPROVED).
//
